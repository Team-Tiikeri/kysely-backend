REST Home Page

All Questionnaires

Method: GET Endpoint: api/questionnaires

JSON Example: 

		
[
  {
    "questionnaireId": 1,
    "title": "Trivia 1",
  },
  {
    "questionnaireId": 2,
    "title": "Trivia 2",
  }
]

Questionnaire by ID

GET api/questionnaires/{id} api/questionnaires/1

JSON Example: 

		
 {
   "questionnaireId": 1,
   "title": "Trivia 1",
 }

Questionnaire questions by ID

GET api/questionnaires/{id}/questions api/questionnaires/1/questions

JSON Example: 

		
[
  {
    "id": 2,
    "type": "text",
    "content": "What is the capital of France?",
    "questionnaire:" {
    	"questionnaireId": 1,
    	"title": "Trivia"
    },
    "required": true 
  },
  {
    "id": 3,
    "type": "text",
    "content": "How long do bears hibernate?",
    "questionnaire:" {
    	"questionnaireId": 1,
    	"title": "Trivia"
    },
    "required": true 
  },
]

Options by question

GET api/questions/{id}/options api/questions/4/options

JSON Example: 

		
{
	question: {
		questionId: 4,
		type: "RADIOBUTTON",
		content: "What is the capital of France?",
		isRequired: true,
		questionnaire: {
			questionnaireId: 1,
			title: "Trivia",
			description: "kinkkinen kysely"
		},
		answers: [
			{
				answerId: 9,
				content: "Paris"
			}
		],
			options: [
				{
					optionId: 10,
					content: "kinkkukiusaus"
				},
			{
				optionId: 11,
				content: "makaronilaatikko"
			},
			{
				optionId: 12,
				content: "lasagne"
			}
		]
	}
}
